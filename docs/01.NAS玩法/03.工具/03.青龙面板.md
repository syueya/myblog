## 1、容器创建

```
services:
  qinglong:
    image: 'whyour/qinglong:latest'
    container_name: qinglong
    restart: always
    network_mode: bridge
    ports:
      - "15700:5700"
    volumes:
      - /volume3/docker/qinglong/data:/ql/data
    environment:
      QlBaseUrl: '/' # 部署路径非必须，以斜杠开头和结尾，比如 /test/
```
