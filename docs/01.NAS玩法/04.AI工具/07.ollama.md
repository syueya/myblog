


## compose部署

```
services:
  ollama:
    image: ollama/ollama:latest
    container_name: ollama
    restart: unless-stopped
    ports:
      - 11434:11434
    volumes:
      - ./data:/root/.ollama # 模型存储目录
    environment:
      - OLLAMA_HOST=0.0.0.0:11434
```
