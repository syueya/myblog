transmission下载器

# 1、compose安装命令

services:
  transmission:
    image: linuxserver/transmission:4.0.5
    container_name: Tr
    network_mode: bridge
    restart: always
    ports:
      - 9091:9091 #网页端口
      - 51414:51413 #Torrent 端口 TCP
      - 51414:51413/udp #Torrent 端口 udp
    volumes:
      - /volume3/docker/transmission/config:/config #配置文件和日志的位置
      - /volume2/Media_movie:/hddMovie
    environment:
      - PUID=0
      - PGID=0
      - TZ=Asia/Shanghai
      - TRANSMISSION_WEB_HOME=/config/web #指定皮肤文件夹的路径
      - USER=admin #web 登陆用户名，可自定义
      - PASS=test1234 #web 登陆密码，可自定义
      - RPCPORT=9091 #web 访问端口号，可自定义，默认为 9091
    deploy:  
      resources:  
        limits:  
          cpus: '4'   # 限制容器可以使用的最大CPU核心数 
          memory: 4G  # 限制容器可以使用的最大内存量
        # reservations:  
        #   cpus: '2'   # 保证容器至少可以使用的CPU核心数
        #   memory: 2G  # 保证容器至少可以使用的内存量
